				jobchange Job_Acolyte_High;
				skill 156,1,0;
				mes "[Father Mareusis]";
				mes "Ну а теперь, вперёд, на помощь голодным и страждущим. С Богом.";

				mes "[Father Mareusis]";
				mes "Вперёд, на помощь голодным и страждущим. С Богом.";
				close;
			}

			mes "[Father Mareusis]";
			mes "Сдаётся мне, ты уже бывал тут.";
			mes "Должно быть, много чего умел до встречи с Валгаллой.";
			next;
			mes "[Father Mareusis]";
			mes "Но тебе явно не сюда. Поройся в кафре, поищи старые вещи и постарайся вспомнить кем ты был раньше. Удачи!";
			close;
		}
	}

	mes "Что ты здесь ищешь?";

	case 1:
		mes "[Father Mareusis]";
		if(BaseJob != Job_Novice) {
			if(BaseJob == Job_Acolyte)
				mes "Are you feeling okay today? I can tell by your attire that you are already an Acolyte. You're not joking around, are you?";
			else
				mes "I'm sorry but it seems you already have your own job, aren't you?";
			close;
		}else{
			mes "Тебе правда охота стать рабом Божьим?";
			mes "Ну давай глянем, готов ты или нет... Ммм...";
			next;
			if (getskilllv("NV_BASIC") < 9) {
				mes "[Father Mareusis]";
				mes "Госпади! Да ты ещё основные навыки не изучил?! Истреби ещё немного порингов и возвращайся как будешь готов!";
				close;
			}
			mes "[Father Mareusis]";
			mes "Ммм... кажется всё в порядке...";
			mes "Хорошо. Сейчас ты преобразишься, это не больно.";
			next;
			mes "[Father Mareusis]";
			mes "Не забывай благодарить всевышнего, он ведь заботится о нас всё время. Храни свою веру несмотря на трудности и окружающий хаос, тебе потом зачтётся.";
			callfunc "Job_Change",Job_Acolyte;
			getitem 1545,1;	// N_Mace
			next;
			mes "[Father Mareusis]";
			mes "И на последок, хочу поздравить с отличным выбором своего пути.";
			close;
		}
	case 2:
		mes "[Father Mareusis]";
		mes "Хочешь стать послушником?";
		mes "Что ж, требования просты.";
		next;
		mes "[Father Mareusis]";
		mes "Первое - нужно достичь профессионального уровня номер 9 и распределить полученный багаж очков в основные навыки.";
		next;
		mes "[Father Mareusis]";
		mes "Это всё. Как посчитаешь себя готовым, приходи.";
		close;
	case 3:
		close;
	}
}

			close;
		}
	}
}
