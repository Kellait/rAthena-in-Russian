hugel,178,142,0	script	toairplane#hugel	45,1,1,{
	end;
OnTouch:
	mes "Чтобы войти на дирижабль, предьявите, пожалуйста, билет на бесплатный перелёт или уплатите 1 200 зен.";
	mes "Желаете совершить посадку?";
	next;
	if (select("Да:Нет") == 1) {
		if (countitem(7311) > 0) {
			delitem 7311,1; //Free_Flying_Ship_Ticket
			warp "airplane",244,58;
			end;
		}
		if (Zeny >= 1200) {
			set Zeny, Zeny - 1200;
			warp "airplane",244,58;
			end;
		}
		mes "К сожалению, у Вас недостаточно денег для посадки. Позвольте напомнить, что посадка стоит 1 200 зен.";
		close;
	}
	mes "Благодарим Вас, счастливого пути.";
	close;
}
